Selenium

Selenium is a powerful open-source framework for automating web applications for testing purposes. It supports multiple browsers and programming languages and enables you to simulate real user interactions with web pages.

This is a Selenium script to fill out a form and validate if the form was submitted.

This is a Selenium script that clicks all links on a webpage and logs broken images.


from selenium import webdriver
from selenium.webdriver.common.by import By 
import time


driver = webdriver.Chrome()                     # Tells selenium which browser we're using, for chrome/firefox add driver path to () eg 'webdriver.Chrome(users/etc/chromedriver)'
driver.get("http://127.0.0.1:5500/index.html#")     # Targets specified website
print(driver.title)   #PRINTS TITLE OF WEBSITE

time.sleep(2)

broken_imgs = []
#------- IMAGE CHECKER -------
def imageChecker():
    images = driver.find_elements(By.TAG_NAME, "img")

    for img in images:
        if img.get_attribute("naturalWidth") == '0':
            broken_imgs.append(img.get_attribute('src'))
            print(f"Broken Image: {img.get_attribute('src')}")
        else:
            print(f"Image Loaded Successfully: {img.get_attribute('src')}")
    print('\n')        
#-----------!


#----------------- CLICKS ALL LINKS ON PAGE

visited_pages = []  # INIT LIST FOR VISITED PAGES

clickable_links = driver.find_elements(By.CSS_SELECTOR, "a")

for link in clickable_links:
    link_name = link.get_attribute('href')
    if link_name not in visited_pages:
        visited_pages.append(link_name)
        print(f'{link_name}')
        driver.implicitly_wait(10)
        link.click()
        driver.implicitly_wait(10)
        imageChecker()
        if len(visited_pages) == 19:        # HAD TO SPECIFY HOW MANY PAGES THERE WERE TO CHECK TO AVOID STALE ELEMENT ERROR
            break
        
print(f'These images didnt load: {broken_imgs}')
